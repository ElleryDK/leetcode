## 3.无重复字符的最长子串
```java
public int lengthOfLongestSubstring(String s) {
    int left = 0,max = 0;
    HashMap<Character,Integer> mp = new HashMap<>();
    for(int i =0;i<s.length();i++){
        if(mp.containsKey(s.charAt(i))){
            left = Math.max(left,mp.get(s.charAt(i))+1);
        }
        max = Math.max(max,i-left+1);
        mp.put(s.charAt(i),i);
    }
    return max;
}
```

## 20.有效的括号
```java
public boolean isValid(String s) {
    Stack<Character> st = new Stack<>();
    for(int i=0;i<s.length();i++){
        if(st.empty()){
            st.push(s.charAt(i));
            continue;
        }
        Character ch = st.peek();
        if(ch=='('&&s.charAt(i)==')')
            st.pop();
        else if(ch=='['&&s.charAt(i)==']')
            st.pop();
        else if(ch=='{'&&s.charAt(i)=='}')
            st.pop();
        else
        st.push(s.charAt(i));
    }
    return st.empty();
}
```

## 125 验证回文串
```java
public boolean isPalindrome(String s) {
    s = s.toLowerCase();
    int len =0;
    Character[] tmp = new Character[s.length()];
    for(int i =0;i<s.length();i++){
        if(s.charAt(i)>='a'&&s.charAt(i)<='z')
            tmp[len++] = s.charAt(i);
        if(s.charAt(i)>='0'&&s.charAt(i)<='9')
            tmp[len++] = s.charAt(i);
    }
    int left = 0,right = len-1;
    while(left<right){
        if(tmp[left]!=tmp[right])
            return false;
        left++;
        right--;
    }
    return true;
}
```

```java
public int countPrimes(int n) {
        int[] record = new int[n+1];
    for(int i =2;i <n;i++){
        for(int j =2;j<=n/i;j++){
            record[i*j] =1;
        }
    }
    int count =0;
    for(int i = 2;i<n;i++){
        if(record[i]==0)
            count++;
    }
    return count;
}
```
