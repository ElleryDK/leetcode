```java
public int lengthOfLongestSubstring(String s) {//指针
    int left = 0,max = 0;
    HashMap<Character,Integer> mp = new HashMap<>();//记录字符出现的位置
    for(int i =0;i<s.length();i++){
        if(mp.containsKey(s.charAt(i))){
            left = Math.max(left,mp.get(s.charAt(i))+1);
        }
        max = Math.max(max,i-left+1);
        mp.put(s.charAt(i),i);
    }
    return max;
}
```

## 20.有效的括号
```java
public boolean isValid(String s) {//栈
    Stack<Character> st = new Stack<>();
    for(int i=0;i<s.length();i++){
        if(st.empty()){
            st.push(s.charAt(i));
            continue;
        }
        Character ch = st.peek();
        if(ch=='('&&s.charAt(i)==')')
            st.pop();
        else if(ch=='['&&s.charAt(i)==']')
            st.pop();
        else if(ch=='{'&&s.charAt(i)=='}')
            st.pop();
        else
        st.push(s.charAt(i));
    }
    return st.empty();
}
```

## 125 验证回文串
```java
public boolean isPalindrome(String s) {
    s = s.toLowerCase();
    int len =0;
    Character[] tmp = new Character[s.length()];
    for(int i =0;i<s.length();i++){//处理输入，去除特殊字符
        if(s.charAt(i)>='a'&&s.charAt(i)<='z')
            tmp[len++] = s.charAt(i);
        if(s.charAt(i)>='0'&&s.charAt(i)<='9')
            tmp[len++] = s.charAt(i);
    }
    int left = 0,right = len-1;
    while(left<right){//左右指针
        if(tmp[left]!=tmp[right])
            return false;
        left++;
        right--;
    }
    return true;
}
```

```java
public int countPrimes(int n) {//去除所有不是质数的数，剩下的是质数
    int[] record = new int[n+1];
    for(int i =2;i <n;i++){
        for(int j =2;j<=n/i;j++){
            record[i*j] =1;
        }
    }
    int count =0;
    for(int i = 2;i<n;i++){
        if(record[i]==0)
            count++;
    }
    return count;
}
```

```java
public String reverseParentheses(String s) {
    Deque<String> stack = new LinkedList<String>();
    StringBuffer sb = new StringBuffer();
    for (int i = 0; i < s.length(); i++) {
        char ch = s.charAt(i);
        if (ch == '(') {
            stack.push(sb.toString());
            sb.setLength(0);
        } else if (ch == ')') {
            sb.reverse();
            sb.insert(0, stack.pop());
        } else {
            sb.append(ch);
        }
    }
    return sb.toString();
}
```
